Install

Usage
import { fig } from 'ai-figure';

const svg = fig(`
  figure flow
  direction: LR
  A[Start] --> B[End]
`);

// Browser
document.getElementById('chart').innerHTML = svg;

// Node.js
import { writeFileSync } from 'fs';
writeFileSync('diagram.svg', svg);
