Usage

  
Get started in seconds

  

    

      
install

      
# npm
npm install ai-flowchart

# pnpm
pnpm add ai-flowchart

    

    

      
basic.ts

      
import { createFlowChart } from 'ai-flowchart';

const svg = createFlowChart({
  nodes: [
    { id: 'start', label: 'Start', type: 'terminal' },
    { id: 'proc',  label: 'Process', type: 'process'  },
    { id: 'end',   label: 'End',     type: 'terminal' },
  ],
  edges: [
    { from: 'start', to: 'proc' },
    { from: 'proc',  to: 'end'  },
  ],
});

    

    

      
browser usage

      
// Inject directly into the page
document.getElementById('chart').innerHTML = svg;

// Or use DOMParser for safe insertion
const parser = new DOMParser();
const doc = parser.parseFromString(svg, 'image/svg+xml');
el.replaceChildren(doc.documentElement);

    

    

      
node.js usage

      
import { writeFileSync } from 'fs';
import { createFlowChart } from 'ai-flowchart';

const svg = createFlowChart({
  nodes: [...],
  edges: [...],
});

writeFileSync('chart.svg', svg);

    

  










  
API Reference

  
createFlowChart(options)


  
FlowNode

  

    
FieldTypeDefaultDescription

    
      
idstring—Unique identifier

      
labelstring—Text shown inside the node

      
typeNodeType'process'process | decision | terminal | io

    
  


  
FlowEdge

  

    
FieldTypeDefaultDescription

    
      
fromstring—Source node ID

      
tostring—Target node ID

      
labelstring—Optional edge label

    
  


  
FlowGroup

  

    
FieldTypeDefaultDescription

    
      
idstring—Unique identifier

      
labelstring—Group label shown above the dashed border

      
nodesstring[]—IDs of nodes inside this group

    
  


  
Options

  

    
FieldTypeDefaultDescription

    
      
nodesFlowNode[]—List of nodes

      
edgesFlowEdge[]—List of directed edges

      
groupsFlowGroup[][]Optional logical groups

      
themeThemeType'excalidraw''excalidraw' | 'clean'

      
directionDirection'TB''TB' top→bottom | 'LR' left→right
